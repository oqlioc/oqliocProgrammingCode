int main(void)
{
	int **stu;
	int n;
	char **name;
	int *sum;
	float *avg;

	printf(" 학생은 몇명? : ");
	scanf("%d", &n);

	stu = (int **)malloc(n * sizeof(int));
	sum = (int *)malloc(n * sizeof(int));

	avg = (float *)malloc(n * sizeof(float));
	name = (char **)malloc(n * sizeof(char*));

	for(int i = 0; i < n; i++)
	{
			name[i] = (char *)malloc(10 * sizeof(char));
			printf(" 학생 이름? : ");
			scanf("%s", name[i]);

			sum[i] = 0;

		for(int j = 0; j < 3; j++)
		{
			stu[i] = (int *)malloc(3 * sizeof(int));

			printf(" %s 학생의 국영수 점수? : " , name[i]);
			scanf("%d", &stu[i][j]);
			sum[i] += stu[i][j];
		}
		avg[i] = sum[i] / 3;
	}


	for(int i = 0; i < n; i++)
	{
		printf(" 이름 : %s \n", name[i]);
		printf(" 총합 : %d \n", sum[i]);
		printf(" 평균 : %f \n", avg[i]);
	}
	
	free(stu);
	free(sum);
	free(avg);
}
