#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>
#include <math.h>

typedef struct student {
	char name[20];
	int kor;
	int eng;
	int math;
	int sum;
	float avg;
}Student ;

void ShowWindow(Student man, int num)
{
	for(int i = 0; i < num; i++)
	{
		printf(" %s 학 생\n", man.name);
		printf("    총 점 : %d\n", man.sum);
		printf("    평 균 : %.3f\n", man.avg);
	}
}

Student Input(int num)
{
	Student man[40];


	for(int i = 0; i < num; i++)
	{
		printf(" 이 름 ? : ");
		scanf("%s", man[i].name);

		printf(" 국 어  영 어  수 학 점수 ? : ");
		scanf("%d %d %d", &man[i].kor, &man[i].eng, &man[i].math);

		man[i].sum = 0;
		
		man[i].sum += man[i].kor + man[i].eng + man[i].math;

		man[i].avg = man[i].sum /3.f;

		return man[i];

	}
}

void main()
{
	
	int num;
