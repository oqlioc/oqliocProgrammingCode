#include <iostream>
using namespace std;
 
 
class Oper {
 
private:
    int num1;
    int num2;
    float res;
    int fournum;
 
public:
    
 
    void scan() {  
        cout << " 정수 입력 ";
        cin >> num1 >> num2;
    }
 
    void fouroper() {
        cout << " 1 : +  ||  2 : -  ||  3 : *  ||  4 : /  ||";
        cin >> fournum;
    
        switch(fournum) {
            case 1 : res = num1 + num2;
                break;
            case 2 : res = num1 - num2;
                break;
            case 3 : res = num1 * num2;
                break;
            case 4 : res = (float)num1 / num2;
                break;
 
            default : cout << " ERR " << endl;
                break;
        }
    }
 
    void print() {
        cout << res << endl;
    }
};
 
void main()
{
    
    Oper *p = new Oper;
 
    p -> scan();
    p -> fouroper();
    p -> print();
}
